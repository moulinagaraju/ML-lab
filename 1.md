import numpy as np
from sklearn.linear_model import LinearRegression
import joblib
# Generate synthetic data for training
np.random.seed(42)
X = np.random.rand(1000, 2) * 100 # Two random numbers
y = X[:, 0] + X[:, 1] # Sum
# Train the model
model = LinearRegression()
model.fit(X, y)
# Save the model
joblib.dump(model, 'model.pkl')
print("Model trained and saved as model.pkl")# ML-lab
